from selenium import webdriver
from selenium.webdriver.common.by import By
import time
import unittest

class Login(unittest.TestCase):

    def setUp(self):
        self.driver = webdriver.Chrome()
        self.driver.get("https://sahyadri.digital/")
        self.driver.maximize_window()

    def test_login(self):
        driver = self.driver
        driver.find_element(By.ID, "student-tab").click()
        time.sleep(4)
        driver.find_element(By.NAME, "admission_no").send_keys("your_admission_no")
        time.sleep(2)
        driver.find_element(By.CLASS_NAME, "student-pass").send_keys("your_password")
        time.sleep(2)
        driver.find_element(By.ID, "login-student").click()
        time.sleep(5)

    def tearDown(self):
        self.driver.quit()

if __name__ == "__main__":
    test = Login()
    test.setUp()
    test.test_login()
    test.tearDown()
